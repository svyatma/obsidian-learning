# Редирект с http на https создан автоматически при заказе бесплатного ssl-сертификата
RewriteEngine On
RewriteCond %{SERVER_PORT} !^443$
RewriteRule .* https://%{SERVER_NAME}%{REQUEST_URI} [R,L]
# Конец редиректа

Options +FollowSymLinks -MultiViews
RewriteEngine On
RewriteBase /

# ----- РЕДИРЕКТ со старого URL (.html) на новый (без .html) -----
RewriteCond %{THE_REQUEST} \s/+(.+?)\.html[\s?] [NC]
RewriteRule ^ /%1 [R=301,L,NE]

# ----- РЕДИРЕКТ если кто-то зашел на .html с параметрами -----
RewriteCond %{THE_REQUEST} \s/+(.+?)\.html\?([^\s]+) [NC]
RewriteRule ^ /%1?%2 [R=301,L,NE]

# ----- ВНУТРЕННЯЯ ПОДСТАНОВКА .html для чистых URL -----
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME}.html -f
RewriteRule ^(.+?)/?$ $1.html [L]

<IfModule mod_expires.c>
  ExpiresActive On
  
  # Кэш на 1 месяц для статических файлов (2592000 секунд = 30 дней)
  ExpiresByType text/css "access plus 1 month"
  ExpiresByType application/javascript "access plus 1 month"
  ExpiresByType application/x-javascript "access plus 1 month"
  ExpiresByType text/javascript "access plus 1 month"
  
  # Изображения
  ExpiresByType image/jpeg "access plus 1 month"
  ExpiresByType image/jpg "access plus 1 month"
  ExpiresByType image/png "access plus 1 month"
  ExpiresByType image/gif "access plus 1 month"
  ExpiresByType image/webp "access plus 1 month"
  ExpiresByType image/svg+xml "access plus 1 month"
  ExpiresByType image/x-icon "access plus 1 month"
  ExpiresByType image/vnd.microsoft.icon "access plus 1 month"
  
  # Шрифты
  ExpiresByType font/ttf "access plus 1 month"
  ExpiresByType font/otf "access plus 1 month"
  ExpiresByType font/woff "access plus 1 month"
  ExpiresByType font/woff2 "access plus 1 month"
  ExpiresByType application/font-woff "access plus 1 month"
  ExpiresByType application/font-woff2 "access plus 1 month"
  
  # Другие статические файлы
  ExpiresByType application/pdf "access plus 1 month"
  ExpiresByType application/json "access plus 1 month"
  ExpiresByType application/xml "access plus 1 month"
</IfModule>

<IfModule mod_headers.c>
  # Добавляем заголовки Cache-Control для надежности
  <FilesMatch "\.(css|js|jpe?g|png|gif|webp|svg|ico|ttf|otf|woff2?|pdf|json|xml)$">
    Header set Cache-Control "max-age=2592000, public, immutable"
  </FilesMatch>
</IfModule>

# Сжимаем файлы для дополнительной экономии (опционально)
<IfModule mod_deflate.c>
  AddOutputFilterByType DEFLATE text/css text/javascript application/javascript application/x-javascript text/html text/plain text/xml application/json image/svg+xml
</IfModule>
